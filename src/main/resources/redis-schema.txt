FT.CREATE deploymentIndex
    ON JSON
    PREFIX 1 deployment:
SCHEMA
    $.id as id TEXT
    $.containerId as containerId TEXT
    $.port as port NUMERIC NOINDEX
    $.dnsRecordId as dnsRecordId TEXT NOINDEX
    $.mode as mode TEXT NOINDEX
    $.host as host TEXT
    $.state as state TEXT NOINDEX
    $.directory as directory TEXT NOINDEX
    $.createdAt as createdAt NUMERIC // milliseconds since unix-epoch
